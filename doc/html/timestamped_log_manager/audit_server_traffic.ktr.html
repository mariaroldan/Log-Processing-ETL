This job entry does not have a description.

Script 1
var lastInputRowIdx = getInputRowMeta().size();

var logTimestampTimeZone = java.util.TimeZone.getTimeZone(getVariable("logTimestampTimeZone", "PST8PDT"));
var gmtTimeZone = java.util.TimeZone.getTimeZone("GMT");

var sdf = new java.text.SimpleDateFormat("yyyy-MM-dd-HH");
sdf.setTimeZone(logTimestampTimeZone);

// Get the current time truncated to hour of day
var cal = java.util.Calendar.getInstance(gmtTimeZone);
cal.set(java.util.Calendar.MINUTE, 0);
cal.set(java.util.Calendar.SECOND, 0);
cal.set(java.util.Calendar.MILLISECOND, 0);

// Subtract one hour for the most recently completed hour
cal.add(java.util.Calendar.HOUR_OF_DAY, -1);
// Subtract one more hour to prevent finding possibly incomplete files that just rolled over
cal.add(java.util.Calendar.HOUR_OF_DAY, -1);
var lastCompleteHour = cal.clone();

// Subtract the allowed lag hours for logs to appear
cal.add(java.util.Calendar.HOUR_OF_DAY, str2num(getVariable("allowedLogLagHours", "-2")));
var hourLogsRequired = cal.clone();

// Get beginning of day, GMT as of 8 hours ago (in case we are late processing)
cal.add(java.util.Calendar.HOUR_OF_DAY, -8);
cal.set(java.util.Calendar.HOUR_OF_DAY, 0);

while (cal.compareTo(lastCompleteHour) <= 0)
{
	var idx = lastInputRowIdx;
	var required = cal.compareTo(hourLogsRequired) <= 0;
	var timestamp = sdf.format(cal.getTime());
	writeToLog("r", "Current row date-hour: "+timestamp);

	newRow = createRowCopy(getOutputRowMeta().size());
	newRow[idx++] = timestamp;
	newRow[idx++] = required;
	putRow(newRow);

	cal.add(java.util.Calendar.HOUR_OF_DAY, 1);
}


trans_Status = SKIP_TRANSFORMATION;
