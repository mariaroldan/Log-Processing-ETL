Set Successful flag for EOD

SQL

SQL

REPLACE INTO eod_status SET `utc_date_string` = '${PREVIOUS_UTC_DATE}', `status` = 'Successful', `audit_snapshot` = (
	SELECT CONCAT(site,'\t',servers,'\t',timestamps) AS results
	FROM (
		SELECT
		  site
		, GROUP_CONCAT(server ORDER BY server SEPARATOR ', ') AS servers
		, timestamps
		FROM (
			SELECT
			  site
			, CONCAT(datacenter,'-',server) AS server
			, GROUP_CONCAT(timestamp ORDER BY timestamp SEPARATOR ', ') AS timestamps
			FROM `server_timestamps_to_process`
			WHERE required = 'Y'
			GROUP BY 1,2
		) a
		GROUP BY 1,3
	) b
	ORDER BY site, timestamps
	LIMIT 1
);
