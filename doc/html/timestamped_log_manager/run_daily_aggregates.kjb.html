INSERT /*+ direct */ INTO amo_download_requests_by_day
SELECT
  utc_date_id
, product
, file_id
, addon_id
, addon_guid
, addon_version
, addon_name
, location_id
, user_agent_os_id
, user_agent_browser_id
, user_agent_locale_id
, request_locale_id
, CASE WHEN x.src IS NULL THEN NULL WHEN v.name IS NULL THEN 'invalid' ELSE x.src END AS source
, COUNT(1) AS requests
FROM amo_download_requests_raw x
LEFT JOIN amo_download_valid_sources v ON x.src IS NOT NULL AND ((v.type = 'full' AND v.name = x.src) OR (v.type = 'prefix' AND LEFT(x.src, LENGTH(v.name)) = v.name))
WHERE utc_date = '${PREVIOUS_UTC_DATE}'
GROUP BY
  utc_date
, utc_date_id
, product
, source
, request_locale_id
, user_agent_locale_id
, user_agent_os_id
, addon_id
, addon_guid
, addon_name
, addon_version
, file_id
, user_agent_browser_id
, location_id;
COMMIT;
