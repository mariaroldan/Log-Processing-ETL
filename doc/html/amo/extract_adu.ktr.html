Table input

This job entry does not have a description.

SQL

SELECT
  TO_DATE(date, 'yyyy-MM-dd') AS "date"
, product_name
, p.formatted_version_major AS product_version
, SUM(requests) AS adu_count
FROM blocklist_requests_by_day b
JOIN dates d ON d.date_id = b.utc_date_id
JOIN products p ON p.product_id = b.product_id
JOIN operating_systems o ON o.os_id = b.os_id
WHERE date ${DATE}
AND p.formatted_version_major != 'Other'
GROUP BY 1, 2, 3
ORDER BY 1, 2, 3
