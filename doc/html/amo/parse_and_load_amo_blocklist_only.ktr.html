import java.util.*;

private static final long dateDimensionEpoch = 946684800543L;

// Note that Kettle UDJCs do not appear to support generics.
private static final Map timeZones = Collections.synchronizedMap(new HashMap());
private static final Map months;
static
{
    Map tmp = new HashMap();
    tmp.put("Jan", new Integer(0));
    tmp.put("Feb", new Integer(1));
    tmp.put("Mar", new Integer(2));
    tmp.put("Apr", new Integer(3));
    tmp.put("May", new Integer(4));
    tmp.put("Jun", new Integer(5));
    tmp.put("Jul", new Integer(6));
    tmp.put("Aug", new Integer(7));
    tmp.put("Sep", new Integer(8));
    tmp.put("Oct", new Integer(9));
    tmp.put("Nov", new Integer(10));
    tmp.put("Dec", new Integer(11));
    months = Collections.unmodifiableMap(tmp);
}

private final Calendar calIn = Calendar.getInstance();
private final Calendar calOut = Calendar.getInstance();
private TimeZone timeZone;
private String lastTimeZoneString;

public boolean processRow(StepMetaInterface smi, StepDataInterface sdi) throws KettleException
{
    Object[] r = getRow();
    if (r == null) {
        setOutputDone();
        return false;
    }

    if (first) {
        first = false;
        calOut.setTimeZone(TimeZone.getTimeZone("GMT"));
        // Workaround for http://jira.pentaho.com/browse/PDI-4981 (which seems to have returned in 4.3)
        data.inputRowMeta = getInputRowMeta();
        data.outputRowMeta = data.inputRowMeta.clone();
        meta.getFields(data.outputRowMeta, getStepname(), null, null, parent);
    }

    Object[] outputRow = RowDataUtil.resizeArray(r, data.outputRowMeta.size());

    if (Boolean.TRUE.equals(get(Fields.In, "result").getObject(r))) {
        String url = get(Fields.In, "url").getString(r);
        if (url != null && url.contains("\\.")) {
            get(Fields.Out, "url").setValue(outputRow, url.replace("\\.", "."));
        }
        String url_params = get(Fields.In, "url_params").getString(r);
        if (url_params != null && url_params.contains("\\.")) {
            get(Fields.Out, "url_params").setValue(outputRow, url_params.replace("\\.", "."));
        }
        String user_agent = get(Fields.In, "user_agent").getString(r);
        if (user_agent != null && user_agent.contains("\\.")) {
            get(Fields.Out, "user_agent").setValue(outputRow, user_agent.replace("\\.", "."));
        }

        String date = get(Fields.In, "date").getString(r);

        int dom = Integer.parseInt(date.substring(0, 2));
        int month = ((Integer)months.get(date.substring(3, 6))).intValue();
        int year = Integer.parseInt(date.substring(7, 11));
        int hour = Integer.parseInt(date.substring(12, 14));
        int minute = Integer.parseInt(date.substring(15, 17));

        String tzs = date.substring(21, 26);
        if (!tzs.equals(lastTimeZoneString)) {
            lastTimeZoneString = tzs;
            timeZone = (TimeZone)timeZones.get(tzs);
            if (timeZone == null) {
                timeZone = TimeZone.getTimeZone("GMT"+tzs);
                timeZones.put(tzs, timeZone);
            }
        }
        
        calIn.setTimeZone(timeZone);
        calIn.set(year, month, dom, hour, minute, 0);
        calOut.setTime(calIn.getTime());

        Long domParse = Long.valueOf(calOut.get(Calendar.DAY_OF_MONTH));
        Long monthParse = Long.valueOf(calOut.get(Calendar.MONTH)+1);
        Long yearParse = Long.valueOf(calOut.get(Calendar.YEAR));

        String dateString = String.format("%d-%02d-%02d", new Object[] { yearParse, monthParse, domParse });
        get(Fields.Out, "utc_date").setValue(outputRow, dateString);
        get(Fields.Out, "utc_hour").setValue(outputRow, Long.valueOf(calOut.get(Calendar.HOUR_OF_DAY)));
        get(Fields.Out, "utc_minute").setValue(outputRow, Long.valueOf(calOut.get(Calendar.MINUTE)));
        get(Fields.Out, "utc_date_id").setValue(outputRow, Long.valueOf((calOut.getTimeInMillis() - dateDimensionEpoch) / (1000 * 60 * 60 * 24) + 1));

        putRow(data.outputRowMeta, outputRow);
    } else {
        incrementLinesRejected();
    }
    return true;
}

import java.util.*;

private static final long JAN_1_11_DATE_ID = 4019L;

public boolean processRow(StepMetaInterface smi, StepDataInterface sdi) throws KettleException
{
    Object[] r = getRow();
    if (r == null) {
        setOutputDone();
        return false;
    }

    if (first) {
        first = false;
        // Workaround for http://jira.pentaho.com/browse/PDI-4981 (which seems to have returned in 4.3)
        data.inputRowMeta = getInputRowMeta();
        data.outputRowMeta = data.inputRowMeta.clone();
        meta.getFields(data.outputRowMeta, getStepname(), null, null, parent);
    }

    String utc_date_string = get(Fields.In, "utc_date_id").getString(r);
    long utc_date_id = -1l;
    try {
       utc_date_id = Long.parseLong(utc_date_string);
    } catch (NumberFormatException nfe) {
       System.out.println("ERROR parsing '" + utc_date_string + "' as a Long. defaulting to -1");
    }

    Object[] out = RowDataUtil.resizeArray(r, data.outputRowMeta.size());

    long pingCount, totalPingCount, daysSinceLastPing;
    String pingCountStr = get(Fields.In, "ping_count_raw").getString(r);
    String totalPingCountStr = get(Fields.In, "total_ping_count_raw").getString(r);
    String daysSinceLastPingStr = get(Fields.In, "days_since_last_ping_raw").getString(r);

    if (pingCountStr == null) {
        get(Fields.Out, "ping_count").setValue(out, null);
    } else {
        if ("invalid".equals(pingCountStr)) {
            pingCount = -1L;
        } else {
            try {
                pingCount = Long.parseLong(pingCountStr);
                if (pingCount < 1L) pingCount = -1L;
                else if (utc_date_id - pingCount < JAN_1_11_DATE_ID) pingCount = -1L;
            } catch (Exception e) {
                pingCount = -1L;
            }
        }
        get(Fields.Out, "ping_count").setValue(out, new Long(pingCount));
    }

    if (totalPingCountStr == null) {
        get(Fields.Out, "total_ping_count").setValue(out, null);
    } else {
        if ("invalid".equals(totalPingCountStr)) {
            totalPingCount = -1L;
        } else {
            try {
                totalPingCount = Long.parseLong(totalPingCountStr);
                if (totalPingCount < 1L) totalPingCount = -1L;
                else if (utc_date_id - totalPingCount < JAN_1_11_DATE_ID) totalPingCount = -1L;
            } catch (Exception e) {
                totalPingCount = -1L;
            }
        }
        get(Fields.Out, "total_ping_count").setValue(out, new Long(totalPingCount));
    }

    if (daysSinceLastPingStr == null) {
        get(Fields.Out, "days_since_last_ping").setValue(out, null);
    } else {
        if ("invalid".equals(daysSinceLastPingStr)) {
            daysSinceLastPing = -1L;
        } else if ("new".equals(daysSinceLastPingStr)) {
            daysSinceLastPing = 365000L;
        } else if ("reset".equals(daysSinceLastPingStr)) {
            daysSinceLastPing = -2L;
        } else {
            try {
                daysSinceLastPing = Long.parseLong(daysSinceLastPingStr);
                if (daysSinceLastPing < 0L) daysSinceLastPing = -1L;
                else if (utc_date_id - daysSinceLastPing < JAN_1_11_DATE_ID) daysSinceLastPing = -1L;
            } catch (Exception e) {
                daysSinceLastPing = -1L;
            }
        }
        //System.err.printf("\r%5s      %5d      ", new Object[] { daysSinceLastPingStr, daysSinceLastPing });
        get(Fields.Out, "days_since_last_ping").setValue(out, new Long(daysSinceLastPing));
    }

    putRow(data.outputRowMeta, out);
    return true;
}
