import java.util.*;

private static final long dateDimensionEpoch = 946684800543L;
private static final Map timeZones = Collections.synchronizedMap(new HashMap());
private static final Map months;
static
{
    Map tmp = new HashMap();
    tmp.put("Jan", new Integer(0));
    tmp.put("Feb", new Integer(1));
    tmp.put("Mar", new Integer(2));
    tmp.put("Apr", new Integer(3));
    tmp.put("May", new Integer(4));
    tmp.put("Jun", new Integer(5));
    tmp.put("Jul", new Integer(6));
    tmp.put("Aug", new Integer(7));
    tmp.put("Sep", new Integer(8));
    tmp.put("Oct", new Integer(9));
    tmp.put("Nov", new Integer(10));
    tmp.put("Dec", new Integer(11));
    months = Collections.unmodifiableMap(tmp);
}

public static class Fields {
	public static class In  { public static FieldHelper result, date, url, url_params, user_agent; }
	public static class Out { public static FieldHelper date, hour, minute, utc_date_id; }
}

private final Calendar calIn = Calendar.getInstance();
private final Calendar calOut = Calendar.getInstance();
private TimeZone timeZone;
private String lastTimeZoneString;

public boolean processRow(StepMetaInterface smi, StepDataInterface sdi) throws KettleException
{
	Object[] r = getRow();
	if (r == null) {
		setOutputDone();
		return false;
	}

    if (first)
    {
        first = false;
	    calOut.setTimeZone(TimeZone.getTimeZone("GMT"));
        data.outputRowMeta = getInputRowMeta().clone();
        meta.getFields(data.outputRowMeta, getStepname(), null, null, parent);
		Fields.In.result = new FieldHelper(data.inputRowMeta, "result");
		Fields.In.date = new FieldHelper(data.inputRowMeta, "date");
		Fields.In.url = new FieldHelper(data.inputRowMeta, "url");
		Fields.In.url_params = new FieldHelper(data.inputRowMeta, "url_params");
		Fields.In.user_agent = new FieldHelper(data.inputRowMeta, "user_agent");
		Fields.Out.date = new FieldHelper(data.outputRowMeta, "utc_date");
		Fields.Out.hour = new FieldHelper(data.outputRowMeta, "utc_hour");
		Fields.Out.minute = new FieldHelper(data.outputRowMeta, "utc_minute");
		Fields.Out.utc_date_id = new FieldHelper(data.outputRowMeta, "utc_date_id");
    }

	if (Boolean.TRUE.equals(Fields.In.result.getObject(r))) {

        String url = Fields.In.url.getString(r);
        if (url != null && url.contains("\\.")) {
            Fields.In.url.setValue(r, url.replace("\\.", "."));
        }
        String url_params = Fields.In.url_params.getString(r);
        if (url_params != null && url_params.contains("\\.")) {
            Fields.In.url_params.setValue(r, url_params.replace("\\.", "."));
        }
        String user_agent = Fields.In.user_agent.getString(r);
        if (user_agent != null && user_agent.contains("\\.")) {
            Fields.In.user_agent.setValue(r, user_agent.replace("\\.", "."));
        }

		String date = Fields.In.date.getString(r);

        int dom = Integer.parseInt(date.substring(0, 2));
        int month = ((Integer)months.get(date.substring(3, 6))).intValue();
        int year = Integer.parseInt(date.substring(7, 11));
        int hour = Integer.parseInt(date.substring(12, 14));
        int minute = Integer.parseInt(date.substring(15, 17));

        String tzs = date.substring(21, 26);
        if (!tzs.equals(lastTimeZoneString)) {
			lastTimeZoneString = tzs;
	        timeZone = (TimeZone)timeZones.get(tzs);
			if (timeZone == null) {
            	timeZone = TimeZone.getTimeZone("GMT"+tzs);
            	timeZones.put(tzs, timeZone);
			}
        }
        
        calIn.setTimeZone(timeZone);
        calIn.set(year, month, dom, hour, minute, 0);
        calOut.setTime(calIn.getTime());

        Long domParse = Long.valueOf(calOut.get(Calendar.DAY_OF_MONTH));
        Long monthParse = Long.valueOf(calOut.get(Calendar.MONTH)+1);
        Long yearParse = Long.valueOf(calOut.get(Calendar.YEAR));

		String dateString = String.format("%d-%02d-%02d", new Object[] { yearParse, monthParse, domParse });
		Fields.Out.date.setValue(r, dateString);
		Fields.Out.hour.setValue(r, Long.valueOf(calOut.get(Calendar.HOUR_OF_DAY)));
		Fields.Out.minute.setValue(r, Long.valueOf(calOut.get(Calendar.MINUTE)));
		Fields.Out.utc_date_id.setValue(r, Long.valueOf((calOut.getTimeInMillis() - dateDimensionEpoch) / (1000 * 60 * 60 * 24) + 1));

        putRow(data.outputRowMeta, r);
	} else {
        incrementLinesRejected();
	}
	return true;
}
