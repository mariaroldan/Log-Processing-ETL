Create table locations

This executes the SQL to create table [metrics].[metrics.locations]

SQL

CREATE TABLE locations
(
  location_id INTEGER PRIMARY KEY
, continent_code VARCHAR(2)
, continent_name VARCHAR(13)
, country_code VARCHAR(2)
, country_name VARCHAR(50)
, region_code VARCHAR(2)
, region_name VARCHAR(50)
, city_name VARCHAR(50)
, latitude FLOAT
, longitude FLOAT
, geohash VARCHAR(6)
, insert_date DATE
);
CREATE PROJECTION locations_super
(
  location_id
, continent_code ENCODING RLE
, continent_name ENCODING RLE
, country_code ENCODING RLE
, country_name ENCODING RLE
, region_code
, region_name
, city_name
, latitude
, longitude
, geohash
, insert_date
) AS SELECT
  location_id
, continent_code
, continent_name
, country_code
, country_name
, region_code
, region_name
, city_name
, latitude
, longitude
, geohash
, insert_date
FROM locations
ORDER BY continent_code, continent_name, country_code, country_name
UNSEGMENTED ALL NODES;
